--Timer.txt
--@breif 基础定时器
--@data 2018/3/9

Timer = {
	m_timerType,	--定时器类型

	m_onFinish, 	--完成回调
	m_onHandle,		--方法句柄
	m_targe,		--调用对象

	m_isFinish,		--完成
	m_isPause,		--暂停
	
	m_isScaledTime,	--是否受时间缩放影响（预留）
}

TimeInfo = {
	time,
	deltaTime,
	unscaledDeltaTime,
	unscaledTime,
	
	fixedTime,
	fixedDeltaTime,
	fixedUnscaledDeltaTime,
	fixedUnscaledTime,
}

TIMER_TYPE = {
	FIX = 1,
	LOOP = 2,
}


function Timer:Create(timerType,handle,targe)
	local obj = {}
	setmetatable( obj, { __index = Timer })
	obj:Init(timerType,handle,targe)
	return obj
end

function Timer:Init(timerType,handle,targe)
	self.m_timerType = timerType
	self.m_onFinish = nil
	self.m_onHandle = handle

	self.m_targe = targe
	
	self.m_isFinish = false
	self.m_isPause = true
end


--@brief 暂停
function Timer:Pause()
	self.m_isPause = true
end

--@brief 恢复
function Timer:Recovery()
	self.m_isPause = false
end

--@brief 销毁
function Timer:Finish()
	self.m_isFinish = true
end
